# Test CMakeLists.txt for RSA size test
cmake_minimum_required(VERSION 3.15)
project(RSASizeTest VERSION 1.0)

# Find Dependencies
find_package(cryptopp CONFIG REQUIRED)

# Define the Test Executable
add_executable(RSASizeTest
    test_rsa_size.cpp
    src/wrappers/RSAWrapper.cpp
)

# Configure Include Directories
target_include_directories(RSASizeTest PRIVATE 
    ${CMAKE_CURRENT_SOURCE_DIR}/include
)

# Link Libraries
target_link_libraries(RSASizeTest PRIVATE 
    cryptopp::cryptopp
    ws2_32      # Windows Sockets for networking
    crypt32     # Windows Cryptography API
)

# Set C++ Standard
set_property(TARGET RSASizeTest PROPERTY CXX_STANDARD 17)
set_property(TARGET RSASizeTest PROPERTY CXX_STANDARD_REQUIRED ON)
